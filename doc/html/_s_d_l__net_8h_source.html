1006 SDL_FORCE_INLINE void _SDLNet_Write16(Uint16 value, void* areap)
1007 {
1008 Uint8* area = (Uint8*)areap;
1009 area[0] = (value >> 8) & 0xFF;
1010 area[1] = value & 0xFF;
1011 }

1013 SDL_FORCE_INLINE void _SDLNet_Write32(Uint32 value, void* areap)
1014 {
1015 Uint8* area = (Uint8*)areap;
1016 area[0] = (value >> 24) & 0xFF;
1017 area[1] = (value >> 16) & 0xFF;
1018 area[2] = (value >> 8) & 0xFF;
1019 area[3] = value & 0xFF;
1020 }
